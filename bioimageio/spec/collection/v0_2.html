104class CollectionEntry(Node, extra="allow"):
105    """A valid resource description (RD).
106    The entry RD is based on the collection description itself.
107    Fields are added/overwritten by the content of `rdf_source` if `rdf_source` is specified,
108    and finally added/overwritten by any fields specified directly in the entry.
109    Except for the `id` field, fields are overwritten entirely, their content is not merged!
110    The final `id` for each collection entry is composed of the collection's `id`
111    and the entry's 'sub-'`id`, specified remotely as part of `rdf_source` or superseeded in-place,
112    such that the `final_entry_id = <collection_id>/<entry_id>`"""
113
114    rdf_source: Optional[FileSource] = None
115    """resource description file (RDF) source to load entry from"""
116
117    id: Optional[ResourceId] = None
118    """Collection entry sub id overwriting `rdf_source.id`.
119    The full collection entry's id is the collection's base id, followed by this sub id and separated by a slash '/'."""
120
121    _descr: Optional[EntryDescr] = PrivateAttr(None)
122
123    @property
124    def rdf_update(self) -> Dict[str, YamlValue]:
125        return self.model_extra or {}
126
127    @property
128    def descr(self) -> Optional[EntryDescr]:
129        if self._descr is None:
130            issue_warning(
131                "Collection entry description not set. Is this entry part of a"
132                + " Collection? A collection entry only has its `descr` set if it is part"
133                + " of a valid collection description.",
134                value=None,
135                severity=ALERT,
136            )
137
138        return self._descr

141class CollectionDescr(
142    GenericDescrBase, extra="allow", title="bioimage.io collection specification"
143):
144    """A bioimage.io collection describes several other bioimage.io resources.
145    Note that collections cannot be nested; resources listed under `collection` may not be collections themselves.
146    """
147
148    type: Literal["collection"] = "collection"
149
150    id: Optional[CollectionId] = None
151    """Model zoo (bioimage.io) wide, unique identifier (assigned by bioimage.io)"""
152
153    collection: NotEmpty[List[CollectionEntry]]
154    """Collection entries"""
155
156    @model_validator(mode="after")
157    def finalize_entries(self) -> Self:
158        context = validation_context_var.get()
159        common_entry_content = {
160            k: v
161            for k, v in self.model_dump(mode="json", exclude_unset=True).items()
162            if k not in ("id", "collection")
163        }
164        common_badges = common_entry_content.pop(
165            "badges", None
166        )  # `badges` not valid for model entries
167        base_id: Optional[CollectionId] = self.id
168
169        seen_entry_ids: Dict[str, int] = {}
170
171        for i, entry in enumerate(self.collection):
172            entry_data: Dict[str, Any] = dict(common_entry_content)
173            # set entry specific root as it might be adapted in the presence of an external entry source
174            entry_root = context.root
175            entry_file_name = context.file_name
176
177            if entry.rdf_source is not None:
178                if not context.perform_io_checks:
179                    issue_warning(
180                        "Skipping IO dependent validation (perform_io_checks=False)",
181                        value=entry.rdf_source,
182                        msg_context=dict(i=i),
183                        field=f"collection[{i}]",
184                    )
185                    continue
186
187                external_data = open_bioimageio_yaml(entry.rdf_source)
188                # add/overwrite common collection entry content with external source
189                entry_data.update(external_data.content)
190                entry_root = external_data.original_root
191                entry_file_name = external_data.original_file_name
192
193            # add/overwrite common+external entry content with in-place entry update
194            entry_data.update(entry.rdf_update)
195
196            # also update explicitly specified `id` field data
197            if entry.id is not None:
198                entry_data["id"] = entry.id
199
200            if "id" in entry_data:
201                entry_id = str(entry_data["id"])
202                if (seen_i := seen_entry_ids.get(entry_id)) is not None:
203                    raise ValueError(
204                        f"Dublicate `id` '{entry_data['id']}' in"
205                        + f" collection[{seen_i}]/collection[{i}]"
206                    )
207
208                seen_entry_ids[entry_id] = i
209            else:
210                raise ValueError(f"Missing `id` for entry {i}")
211
212            if base_id is not None:
213                entry_data["id"] = f"{base_id}/{entry_data['id']}"
214
215            type_ = entry_data.get("type")
216            if type_ == "collection":
217                raise ValueError(
218                    f"collection[{i}] has invalid entry type; collections may not be"
219                    + " nested!"
220                )
221
222            if (
223                type_ != "model"
224                and common_badges is not None
225                and "badges" not in entry_data
226            ):
227                # set badges from the collection root for non-model resources if not set for this specific entry
228                entry_data["badges"] = common_badges
229
230            entry_descr = build_description_impl(
231                entry_data,
232                context=context.replace(root=entry_root, file_name=entry_file_name),
233                get_rd_class=partial(
234                    get_rd_class_impl, descriptions_map=_ENTRY_DESCR_MAP
235                ),
236            )
237            assert entry_descr.validation_summary is not None
238            if isinstance(entry_descr, InvalidDescr):
239                raise ValueError(
240                    "Invalid collection entry"
241                    + f" collection[{i}]:\n"
242                    + f"{entry_descr.validation_summary.format(hide_source=True, hide_env=True, root_loc=('collection', i))}"
243                )
244            elif isinstance(
245                entry_descr, get_args(EntryDescr)
246            ):  # TODO: use EntryDescr as union (py>=3.10)
247                entry._descr = entry_descr  # type: ignore
248            else:
249                raise ValueError(
250                    f"{entry_descr.type} {entry_descr.format_version} entries"
251                    + f" are not allowed in {self.type} {self.format_version}."
252                )
253
254        return self

256    @model_validator(mode="before")
257    @classmethod
258    def move_groups_to_collection_field(
259        cls, data: BioimageioYamlContent
260    ) -> BioimageioYamlContent:
261        if data.get("format_version") not in ("0.2.0", "0.2.1"):
262            return data
263
264        if "collection" in data and data["collection"] is not None:
265            if not isinstance(data["collection"], collections.abc.Sequence):
266                raise ValueError(
267                    "Expected `collection` to not be present, or to be a list"
268                )
269
270            data["collection"] = list(data["collection"])
271        else:
272            data["collection"] = []
273
274        for group in ["application", "model", "dataset", "notebook"]:
275            if group in data:
276                data["collection"] += data[group]  # type: ignore
277                data["collection"][-1]["type"] = group
278
279        config = data.get("config")
280        if config and isinstance(config, dict):
281            id_ = config.pop("id", data.get("id"))
282            if id_ is not None:
283                data["id"] = id_
284
285        return data
