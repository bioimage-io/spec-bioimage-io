128    def _get_absolute_impl(
129        self, root: Union[RootHttpUrl, AbsoluteDirectory, pydantic.AnyUrl]
130    ) -> Union[Path, HttpUrl]:
131        if isinstance(root, Path):
132            return (root / self.root).absolute()
133
134        parsed = urlsplit(str(root))
135        path = list(parsed.path.strip("/").split("/"))
136        rel_path = self.root.as_posix().strip("/")
137        if (
138            parsed.netloc == "zenodo.org"
139            and parsed.path.startswith("/api/records/")
140            and parsed.path.endswith("/content")
141        ):
142            path.insert(-1, rel_path)
143        else:
144            path.append(rel_path)
145
146        return HttpUrl(
147            urlunsplit(
148                (
149                    parsed.scheme,
150                    parsed.netloc,
151                    "/".join(path),
152                    parsed.query,
153                    parsed.fragment,
154                )
155            )
156        )

235def validate_suffix(
236    value: V_suffix, suffix: Union[str, Sequence[str]], case_sensitive: bool
237) -> V_suffix:
238    """check final suffix"""
239    if isinstance(suffix, str):
240        suffixes = [suffix]
241    else:
242        suffixes = suffix
243
244    assert len(suffixes) > 0, "no suffix given"
245    assert all(
246        suff.startswith(".") for suff in suffixes
247    ), "expected suffixes to start with '.'"
248    o_value = value
249    strict = interprete_file_source(value)
250
251    if isinstance(strict, (HttpUrl, AnyUrl)):
252        if strict.path is None or "." not in (path := strict.path):
253            actual_suffix = ""
254        elif (
255            strict.host == "zenodo.org"
256            and path.startswith("/api/records/")
257            and path.endswith("/content")
258        ):
259            actual_suffix = "." + path[: -len("/content")].split(".")[-1]
260        else:
261            actual_suffix = "." + path.split(".")[-1]
262
263    elif isinstance(strict, PurePath):
264        actual_suffix = strict.suffixes[-1]
265    elif isinstance(strict, RelativeFilePath):
266        actual_suffix = strict.path.suffixes[-1]
267    else:
268        assert_never(strict)
269
270    if (
271        case_sensitive
272        and actual_suffix not in suffixes
273        or not case_sensitive
274        and actual_suffix.lower() not in [s.lower() for s in suffixes]
275    ):
276        if len(suffixes) == 1:
277            raise ValueError(f"Expected suffix {suffixes[0]}, but got {actual_suffix}")
278        else:
279            raise ValueError(
280                f"Expected a suffix from {suffixes}, but got {actual_suffix}"
281            )
282
283    return o_value

317def _package(value: FileSource, info: SerializationInfo) -> Union[str, Path, FileName]:
318    if (packaging_context := packaging_context_var.get()) is None:
319        # convert to standard python obj
320        # note: pydantic keeps returning Rootmodels (here `HttpUrl`) as-is, but if
321        #   this function returns one RootModel, paths are "further serialized" by
322        #   returning the 'root' attribute, which is incorrect.
323        #   see https://github.com/pydantic/pydantic/issues/8963
324        #   TODO: follow up on https://github.com/pydantic/pydantic/issues/8963
325        if isinstance(value, Path):
326            unpackaged = value
327        elif isinstance(value, HttpUrl):
328            unpackaged = value
329        elif isinstance(value, RelativeFilePath):
330            unpackaged = Path(value.path)
331        elif isinstance(value, AnyUrl):
332            unpackaged = str(value)
333        else:
334            assert_never(value)
335
336        if info.mode_is_json():
337            # convert to json value  # TODO: remove and let pydantic do this?
338            if isinstance(unpackaged, Path):
339                unpackaged = str(unpackaged)
340            elif isinstance(unpackaged, str):
341                pass
342            else:
343                assert_never(unpackaged)
344        else:
345            warnings.warn(
346                "dumping with mode='python' is currently not fully supported for "
347                + "fields that are included when packaging; returned objects are "
348                + "standard python objects"
349            )
350
351        return unpackaged  # return unpackaged file source
352
353    # package the file source:
354    # add it to the current package's file sources and return its collision free file name
355    if isinstance(value, RelativeFilePath):
356        src = value.absolute
357    elif isinstance(value, pydantic.AnyUrl):
358        src = HttpUrl(str(value))
359    elif isinstance(value, HttpUrl):
360        src = value
361    elif isinstance(value, Path):
362        src = value.resolve()
363    else:
364        assert_never(value)
365
366    fname = extract_file_name(src)
367    if fname == packaging_context.bioimageio_yaml_file_name:
368        raise ValueError(
369            f"Reserved file name '{packaging_context.bioimageio_yaml_file_name}' "
370            + "not allowed for a file to be packaged"
371        )
372
373    fsrcs = packaging_context.file_sources
374    assert not any(
375        fname.endswith(special) for special in ALL_BIOIMAGEIO_YAML_NAMES
376    ), fname
377    if fname in fsrcs and fsrcs[fname] != src:
378        for i in range(2, 20):
379            fn, *ext = fname.split(".")
380            alternative_file_name = ".".join([f"{fn}_{i}", *ext])
381            if (
382                alternative_file_name not in fsrcs
383                or fsrcs[alternative_file_name] == src
384            ):
385                fname = alternative_file_name
386                break
387        else:
388            raise ValueError(f"Too many file name clashes for {fname}")
389
390    fsrcs[fname] = src
391    return fname

531def _get_unique_file_name(url: Union[HttpUrl, pydantic.HttpUrl]):
532    """
533    Create a unique file name based on the given URL;
534    adapted from pooch.utils.unique_file_name
535    """
536    md5 = hashlib.md5(str(url).encode()).hexdigest()
537    fname = extract_file_name(url)
538    # Crop the start of the file name to fit 255 characters including the hash
539    # and the :
540    fname = fname[-(255 - len(md5) - 1) :]
541    unique_name = f"{md5}-{fname}"
542    return unique_name

545def download(
546    source: Union[PermissiveFileSource, FileDescr],
547    /,
548    **kwargs: Unpack[HashKwargs],
549) -> DownloadedFile:
550    """download `source` URL (or pass local file path)"""
551    if isinstance(source, FileDescr):
552        return source.download()
553
554    strict_source = interprete_file_source(source)
555    if isinstance(strict_source, RelativeFilePath):
556        strict_source = strict_source.absolute
557
558    if isinstance(strict_source, PurePath):
559        if not strict_source.exists():
560            raise FileNotFoundError(strict_source)
561        local_source = strict_source
562        root: Union[RootHttpUrl, DirectoryPath] = strict_source.parent
563    else:
564        if strict_source.scheme not in ("http", "https"):
565            raise NotImplementedError(strict_source.scheme)
566
567        if settings.CI:
568            headers = {"User-Agent": "ci"}
569            progressbar = False
570        else:
571            headers = {}
572            progressbar = True
573
574        if settings.user_agent is not None:
575            headers["User-Agent"] = settings.user_agent
576
577        downloader = pooch.HTTPDownloader(headers=headers, progressbar=progressbar)
578        fname = _get_unique_file_name(strict_source)
579        _ls: Any = pooch.retrieve(
580            url=str(strict_source),
581            known_hash=_get_known_hash(kwargs),
582            downloader=downloader,
583            fname=fname,
584            path=settings.cache_path,
585        )
586        local_source = Path(_ls).absolute()
587        root = strict_source.parent
588
589    return DownloadedFile(
590        local_source,
591        root,
592        extract_file_name(strict_source),
593    )

652def get_sha256(path: Path) -> Sha256:
653    """from https://stackoverflow.com/a/44873382"""
654    h = hashlib.sha256()
655    b = bytearray(128 * 1024)
656    mv = memoryview(b)
657    with open(path, "rb", buffering=0) as f:
658        for n in iter(lambda: f.readinto(mv), 0):
659            h.update(mv[:n])
660
661    sha = h.hexdigest()
662    assert len(sha) == 64
663    return Sha256(sha)
